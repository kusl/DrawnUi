    strategy:
      matrix:

          - os: macos-latest
            framework: net9.0-ios
            workload: "maui ios"
          - os: macos-latest
            framework: net9.0-maccatalyst
            workload: "maui maccatalyst"

    runs-on: ${{ matrix.os }}
    
    steps:
    - name: Checkout code

      uses: actions/setup-java@v4
